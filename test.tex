\section{The Real and Complex Number Systems}
\label{sec:the_real_and_complex_number_systems}

\begin{thm}
    If $a$ and $b$ are real, then $(a, b) = a + bi$.
\end{thm}

\begin{proof}
    \begin{align*}
        a + bi &= (a, 0) + (b, 0) (0, 1) \\
               &= (a, 0) + (0, b) = (a, b).
    \end{align*}
\end{proof}

\begin{notate}
    If $x_1, \dots, x_n$ are complex numbers, we write
    \[
        x_1 + x_2 + \cdots + \sum_{j = 1}^n x_j.
    \]
\end{notate}
